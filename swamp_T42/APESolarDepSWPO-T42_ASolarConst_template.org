    str =  "#+CAPTION: #{caption}.\n" 
    str << "#+NAME: fig:#{figName} \n" << "|"

    figPicPathStrs = Dir.glob("#{dirPath}#{figPicsPath}")
    

    figPicPathStrs.each{|figPath|
      figPath_ = figPath.gsub("_anim.mp4", "_snapshot.png")

      str << "[[file:#{figPath}][file:#{figThumbPath}]] |" 
    }
    str << "\n"
    return str
